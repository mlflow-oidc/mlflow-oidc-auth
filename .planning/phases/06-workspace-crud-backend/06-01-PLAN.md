    ```python
    def _auto_grant_workspace_manage_permission(response: Response) -> None:
        """Auto-grant MANAGE permission to workspace creator after successful creation (WSCRUD-01)."""
        if not config.MLFLOW_ENABLE_WORKSPACES:
            return
        if response.status_code not in (200, 201):
            return
        auth_context = get_auth_context()
        data = response.get_json(silent=True)
        if not data or "workspace" not in data:
            return
        workspace_info = data["workspace"]
        workspace_name = workspace_info.get("name") if isinstance(workspace_info, dict) else None
        if not workspace_name:
            return
        try:
            store.create_workspace_permission(workspace_name, auth_context.username, "MANAGE")
            flush_workspace_cache()
            get_logger().info(
                f"Auto-granted MANAGE permission on workspace '{workspace_name}' to creator '{auth_context.username}'"
            )
        except Exception:
            # Permission may already exist (race condition) — not an error
            get_logger().debug(
                f"Workspace permission already exists for {auth_context.username} on {workspace_name}"
            )


    def _cascade_delete_workspace_permissions(response: Response) -> None:
        """Cascade-delete all permission rows for a workspace after successful deletion (WSCRUD-06)."""
        if not config.MLFLOW_ENABLE_WORKSPACES:
            return
        if response.status_code not in (200, 204):
            return
        workspace_name = getattr(g, "_deleting_workspace_name", None)
        if not workspace_name:
            return
        try:
            count = store.wipe_workspace_permissions(workspace_name)
            flush_workspace_cache()
            get_logger().info(
                f"Cascade-deleted {count} permission rows for workspace '{workspace_name}'"
            )
        except Exception as e:
            get_logger().warning(f"Failed to cascade-delete permissions for workspace '{workspace_name}': {e}")
    ```
