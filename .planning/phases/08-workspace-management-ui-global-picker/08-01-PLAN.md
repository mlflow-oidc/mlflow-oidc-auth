Backend workspace CRUD router (prefix: `/api/3.0/mlflow/workspaces/crud`):
- POST `/api/3.0/mlflow/workspaces/crud` ŌĆö create workspace (admin only). Body: `{ name: string, description?: string }`. Returns `{ name, description }`. 409 if exists.
- GET `/api/3.0/mlflow/workspaces/crud` ŌĆö list workspaces (filtered by permissions for non-admins). Returns `WorkspaceCrudResponse[]`.
- GET `/api/3.0/mlflow/workspaces/crud/{workspace}` ŌĆö get workspace details. Requires READ permission.
- PATCH `/api/3.0/mlflow/workspaces/crud/{workspace}` ŌĆö update workspace. Body: `{ description: string }`. Requires MANAGE permission.
- DELETE `/api/3.0/mlflow/workspaces/crud/{workspace}` ŌĆö delete workspace. Requires MANAGE permission. 400 if "default". 409 if not empty.

  WorkspaceCrudResponse,
  WorkspaceUserPermission,
  WorkspaceGroupPermission,
  WorkspaceMemberCounts,

export async function createWorkspace(data: WorkspaceCrudCreateRequest): Promise<WorkspaceCrudResponse> {
  return request<WorkspaceCrudResponse>(STATIC_API_ENDPOINTS.WORKSPACE_CRUD, {
    method: "POST",
    body: JSON.stringify(data),

export async function updateWorkspace(workspace: string, data: WorkspaceCrudUpdateRequest): Promise<WorkspaceCrudResponse> {
  return request<WorkspaceCrudResponse>(DYNAMIC_API_ENDPOINTS.WORKSPACE_CRUD_DETAIL(workspace), {

export async function deleteWorkspace(workspace: string): Promise<void> {

  const [users, groups] = await Promise.all([
    fetchWorkspaceUsers(workspace, signal),
    fetchWorkspaceGroups(workspace, signal),
  ]);
  return { users: users.length, groups: groups.length };
}

Updated hook signature:
```typescript
export function useAllWorkspaces() {
  // existing fetch of workspaces...
  const [memberCounts, setMemberCounts] = useState<Record<string, WorkspaceMemberCounts> | null>(null);

  useEffect(() => {
    if (!allWorkspaces?.length) return;
    const controller = new AbortController();
    Promise.all(
      allWorkspaces.map(async (ws) => {
        const counts = await fetchWorkspaceMemberCounts(ws.name, controller.signal);
        return [ws.name, counts] as const;
      })
    ).then((results) => {
      setMemberCounts(Object.fromEntries(results));
    }).catch(() => { /* ignore abort errors */ });
    return () => controller.abort();
  }, [allWorkspaces]);

  return { allWorkspaces, memberCounts, isLoading, error, refresh };
}
```
